namespace = utopia

# Abducted Scientist
country_event = {
	id = utopia.62
	title = "utopia.62.name"
	desc = "utopia.62.desc"
	picture = GFX_evt_think_tank
	show_sound = event_administrative_work
	location = event_target:think_tank_2

	is_triggered_only = yes

	trigger = {
		any_owned_leader = {
			leader_class = scientist
			is_councilor = yes
			is_ruler = no
			NOR = {
				has_leader_flag = leader_death_events_blocked
				has_leader_flag = legendary_leader
				has_leader_flag = renowned_leader
			}
		}
	}

	immediate = {
		random_owned_leader = {
			limit = {
				leader_class = scientist
				is_councilor = yes
				is_ruler = no
				NOR = {
					has_leader_flag = leader_death_events_blocked
					has_leader_flag = legendary_leader
					has_leader_flag = renowned_leader
				}
			}
			save_event_target_as = abducted_leader
			exile_leader_as = exiled_physics_leader
			add_experience = 400
			if = {
				limit = {
					NOR = {
					has_trait = leader_trait_spark_of_genius
					has_trait = leader_trait_spark_of_genius_2
					}
				}
				add_trait = leader_trait_spark_of_genius
			}
			if = {
				limit = {
					NOT = { has_trait_tier1or2or3or4 = { TRAIT = leader_trait_maniacal } }
				}
				add_trait = leader_trait_maniacal
			}
		}
		set_country_flag = think_tank_event
	}
	option = {
		name = utopia.62.a
		hidden_effect = { country_event = { id = utopia.63 days = 360 random = 20 } }
	}
}
